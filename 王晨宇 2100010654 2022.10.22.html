def match(data):
    data0=data.dropna()
    par_norm=stats.norm.fit(data0)

def vis_month(ts_code):
    img=data_monthly[ts_code]
    n=img.shape[0]
    img=DataFrame(img)
    img.dropna(inplace=True)
    img['values']=img.rank()/img.shape[0]
    plt.xlabel='earning rate'
    plt.ylabel='distribution'
    plt.plot(img[ts_code],img['values'],'o',label='original data')
    x=np.arange(-0.4,1,0.001)
    def residuals_norm(p):
        loc,scale=p
        return img['values']-stats.norm.cdf(img[ts_code],loc,scale)
    r_norm=optimize.leastsq(residuals_norm,[0,1])
    plt.plot(x,stats.norm.cdf(x,r_norm[0][0],r_norm[0][1]),label='norm')
    def residuals_t(p):
        df,loc,scale=p
        return img['values']-stats.t.cdf(img[ts_code],df,loc,scale)

    def residuals_gennorm(p):
        beta,loc,scale=p

    plt.title(ts_code+' Monthly Rate')
    plt.legend()
